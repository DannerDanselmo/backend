package com.exemplo.recomendador.controller;

import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class RecomendadorController {

    @GetMapping("/recomendar")
    public String recomendar(
            @RequestParam String clima,
            @RequestParam String estilo) {

        if (clima.equalsIgnoreCase("calor") && estilo.equalsIgnoreCase("natureza")) {
            return "Rio de Janeiro";
        } else if (clima.equalsIgnoreCase("frio") && estilo.equalsIgnoreCase("montanha")) {
            return "Gramado";
        } else if (clima.equalsIgnoreCase("calor") && estilo.equalsIgnoreCase("praia")) {
            return "Salvador";
        } else if (clima.equalsIgnoreCase("frio") && estilo.equalsIgnoreCase("cidade")) {
            return "Curitiba";
        } else {
            return "Destino não encontrado, cupixa 😅";
        }
    }
}

2. Service (JogadorService)

package com.exemplo.jogador.service;

import com.exemplo.jogador.model.Jogador;
import org.springframework.stereotype.Service;

import java.util.Random;

@Service
public class JogadorService {

    private String[] nomes = {"Ronaldo", "Neymar", "Gabriel", "Lucas", "Pedro"};
    private String[] sobrenomes = {"Silva", "Souza", "Fenômeno", "Oliveira", "Santos"};
    private String[] posicoes = {"Goleiro", "Zagueiro", "Meio-campo", "Atacante"};

    public Jogador gerarJogador(String time, String posicao) {
        Random random = new Random();

        String nome = nomes[random.nextInt(nomes.length)];
        String sobrenome = sobrenomes[random.nextInt(sobrenomes.length)];
        int idade = 18 + random.nextInt(20); // idade entre 18 e 37

        // usa a posição recebida ou sorteia se vier inválida
        String posicaoFinal = posicao != null ? posicao : posicoes[random.nextInt(posicoes.length)];

        return new Jogador(
                nome,
                sobrenome,
                idade,
                posicaoFinal,
                capitalize(time)
        );
    }

    private String capitalize(String texto) {
        if (texto == null || texto.isEmpty()) return texto;
        return texto.substring(0,1).toUpperCase() + texto.substring(1).toLowerCase();
    }
}

package com.exemplo.estudantes.service;

import com.exemplo.estudantes.model.Estudante;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;

@Service
public class EstudanteService {

    private List<Estudante> estudantes = new ArrayList<>();

    public Estudante salvar(Estudante estudante) {
        estudantes.add(estudante);
        return estudante;
    }

    public List<Estudante> listarTodos() {
        return estudantes;
    }

    public Estudante buscarPorCodigo(int codigo) {
        return estudantes.stream()
                .filter(e -> e.getCodigo() == codigo)
                .findFirst()
                .orElse(null);
    }
}

package com.exemplo.playlist.service;

import com.exemplo.playlist.model.Musica;
import com.exemplo.playlist.model.Playlist;
import org.springframework.stereotype.Service;

import java.util.*;

@Service
public class PlaylistService {

    private Map<Integer, Playlist> playlists = new HashMap<>();
    private int contadorPlaylist = 1;
    private int contadorMusica = 1;

    // Criar playlist
    public int criarPlaylist(Playlist playlist) {
        playlist.setId(contadorPlaylist++);
        playlists.put(playlist.getId(), playlist);
        return playlist.getId();
    }

    // Adicionar música
    public String adicionarMusica(int playlistId, Musica musica) {
        Playlist playlist = playlists.get(playlistId);

        if (playlist == null) {
            return "Playlist não encontrada";
        }

        musica.setId(contadorMusica++);
        playlist.getMusicas().add(musica);

        return "Música adicionada com sucesso";
    }

    // Listar playlists
    public List<Playlist> listarPlaylists() {
        return new ArrayList<>(playlists.values());
    }

    // Listar músicas de uma playlist
    public List<Musica> listarMusicas(int playlistId) {
        Playlist playlist = playlists.get(playlistId);

        if (playlist == null) {
            return null;
        }

        return playlist.getMusicas();
    }
}

package com.exemplo.playlist.controller;

        return playlistService.criarPlaylist(playlist);
    }

    @PostMapping("/{id}/musicas")
    public String adicionarMusica(
            @PathVariable int id,
            @RequestBody Musica musica) {

        return playlistService.adicionarMusica(id, musica);
    }

    // GET → listar playlists
    @GetMapping
    public List<Playlist> listarPlaylists() {
        return playlistService.listarPlaylists();
