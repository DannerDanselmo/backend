public class Fatura {
    String numero;
    String descricao;
    Integer quantidade;
    Double precoPorItem;

    public Fatura(String numero, String descricao, Integer quantidade, Double precoPorItem) {
        this.numero = numero;
        this.descricao = descricao;
        this.quantidade = quantidade;
        this.precoPorItem = precoPorItem;
    }
    public Double totalFatura(){
        return this.quantidade * this.precoPorItem;
    }
}
