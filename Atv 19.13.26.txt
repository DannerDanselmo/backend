import java.util.ArrayList;
import java.util.List;
import java.util.Random;

class Personagem {
    String nome;
    int vida;
    int ataque;

    public Personagem(String nome, int vida, int ataque) {
        this.nome = nome;
        this.vida = vida;
        this.ataque = ataque;
    }

    public void receberDano(int dano) {
        if (vida <= 0) {
            return;
        }

        vida -= dano;

        if (vida <= 0) {
            vida = 0;
            System.out.println(nome + " foi derrotado!");
        } else {
            System.out.println(nome + " agora tem " + vida + " de vida.");
        }
    }

    public void atacar(Personagem alvo) {
        if (this.vida <= 0) {
            return;
        }

        if (alvo.vida <= 0) {
            System.out.println(this.nome + " tentou atacar " + alvo.nome + ", mas ele já está derrotado!");
            return;
        }

        System.out.println(this.nome + " atacou " + alvo.nome + " causando " + this.ataque + " de dano!");
        alvo.receberDano(this.ataque);
    }

    public boolean estaVivo() {
        return vida > 0;
    }
}

public class Main {
    public static void main(String[] args) {

        List<Personagem> inimigos = new ArrayList<>();
        inimigos.add(new Personagem("Orc", 100, 10));
        inimigos.add(new Personagem("Goblin", 50, 5));
        inimigos.add(new Personagem("Lobo", 75, 15));
        inimigos.add(new Personagem("Cavaleiro", 200, 20));
        inimigos.add(new Personagem("Dragão", 500, 50));

        Personagem jogador = new Personagem("Steve", 100, 20);

        Random random = new Random();
        int rodada = 1;

        while (jogador.estaVivo() && existeInimigoVivo(inimigos)) {

            System.out.println("\n===== RODADA " + rodada + " =====");

            // Mostrar status dos inimigos
            System.out.println("Inimigos:");
            for (Personagem inimigo : inimigos) {
                System.out.println("- " + inimigo.nome + " | Vida: " + inimigo.vida);
            }

            // Jogador ataca
            Personagem alvo = escolherInimigoVivo(inimigos, random);
            if (alvo != null) {
                jogador.atacar(alvo);
            }

            // Inimigos atacam
            System.out.println("\nTurno dos inimigos:");
            for (Personagem inimigo : inimigos) {
                if (inimigo.estaVivo()) {
                    inimigo.atacar(jogador);
                }
            }

            rodada++;
        }

        System.out.println("\n===== FIM DA BATALHA =====");

        if (!jogador.estaVivo()) {
            System.out.println("Steve foi derrotado!");
        } else {
            System.out.println("Todos os inimigos foram derrotados!");
        }
    }

    public static boolean existeInimigoVivo(List<Personagem> inimigos) {
        for (Personagem p : inimigos) {
            if (p.estaVivo()) {
                return true;
            }
        }
        return false;
    }

    public static Personagem escolherInimigoVivo(List<Personagem> inimigos, Random random) {
        List<Personagem> vivos = new ArrayList<>();

        for (Personagem p : inimigos) {
            if (p.estaVivo()) {
                vivos.add(p);
            }
        }

        if (vivos.isEmpty()) {
            return null;
        }

        return vivos.get(random.nextInt(vivos.size()));
    }
}

import java.util.ArrayList;
import java.util.List;

// Classe Barco
class Barco {
    String nome;
    int tamanho;

    public Barco(String nome, int tamanho) {
        this.nome = nome;
        this.tamanho = tamanho;
    }
}

// Classe BasePorto
class BasePorto {
    String nome;
    List<Barco> barcosAtracados;

    public BasePorto(String nome) {
        this.nome = nome;
        this.barcosAtracados = new ArrayList<>();
    }

    public boolean atracarBarco(Barco barco) {
        barcosAtracados.add(barco);
        return true;
    }

    public void desatracarBarco(Barco barco) {
        barcosAtracados.remove(barco);
    }
}

// Porto Pequeno
class PortoPequeno extends BasePorto {

    public PortoPequeno(String nome) {
        super(nome);
    }

    @Override
    public boolean atracarBarco(Barco barco) {
        if (barco.tamanho <= 10) {
            barcosAtracados.add(barco);
            System.out.println(barco.nome + " atracado no porto pequeno");
            return true;
        }
        return false;
    }
}

// Porto Grande
class PortoGrande extends BasePorto {

    public PortoGrande(String nome) {
        super(nome);
    }

    @Override
    public boolean atracarBarco(Barco barco) {
        if (barco.tamanho >= 10) {
            barcosAtracados.add(barco);
            System.out.println(barco.nome + " atracado no porto grande");
            return true;
        }
        return false;
    }
}

// Classe principal
public class Main {
    public static void main(String[] args) {

        // Lista de barcos
        List<Barco> barcos = new ArrayList<>();
        barcos.add(new Barco("Barco 1", 2));
        barcos.add(new Barco("Barco 2", 5));
        barcos.add(new Barco("Barco 3", 7));
        barcos.add(new Barco("Barco 4", 12));
        barcos.add(new Barco("Barco 5", 20));

        // Portos
        PortoPequeno portoPequeno = new PortoPequeno("Porto Pequeno");
        PortoGrande portoGrande = new PortoGrande("Porto Grande");

        // Lógica de atracagem
        for (Barco barco : barcos) {
            boolean atracou = portoPequeno.atracarBarco(barco);

            if (!atracou) {
                portoGrande.atracarBarco(barco);
            }
        }
    }
}

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

// Classe Musica
class Musica {
    String titulo;
    String artista;

    public Musica(String titulo, String artista) {
        this.titulo = titulo;
        this.artista = artista;
    }
}

// Classe Playlist
class Playlist {
    String nome;
    List<Musica> musicas;

    public Playlist(String nome) {
        this.nome = nome;
        this.musicas = new ArrayList<>();
    }

    public void adicionarMusica(Musica musica) {
        musicas.add(musica);
        System.out.println("Você adicionou a música " + musica.titulo + " na playlist.");
    }

    public void mostrarPlaylist() {
        System.out.println("\nPlaylist:");
        for (Musica m : musicas) {
            System.out.println("- " + m.titulo);
        }
    }
}

// Classe MusicCloud
class MusicCloud {
    List<Musica> musicasDisponiveis;

    public MusicCloud() {
        musicasDisponiveis = new ArrayList<>();

        // Músicas disponíveis
        musicasDisponiveis.add(new Musica("Hino do Real Paulista", "Desconhecido"));
        musicasDisponiveis.add(new Musica("Macarena", "Los Del Rio"));
        musicasDisponiveis.add(new Musica("Evidências", "Chitãozinho & Xororó"));
    }

    public Musica buscarMusica(String nome) throws Exception {
        for (Musica m : musicasDisponiveis) {
            if (m.titulo.equalsIgnoreCase(nome)) {
                return m;
            }
        }
        throw new Exception("A música " + nome + " não foi encontrada no sistema.");
    }
}

// Classe principal
public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        MusicCloud cloud = new MusicCloud();

        // Nome da playlist
        System.out.print("Digite o nome da playlist: ");
        String nomePlaylist = sc.nextLine();

        Playlist playlist = new Playlist(nomePlaylist);

        String continuar;

        do {
            try {
                System.out.print("\nDigite o nome da música: ");
                String nomeMusica = sc.nextLine();

                Musica musica = cloud.buscarMusica(nomeMusica);
                playlist.adicionarMusica(musica);

            } catch (Exception e) {
                System.out.println(e.getMessage());
            }

            System.out.print("Deseja adicionar outra música? (s/n): ");
            continuar = sc.nextLine();

        } while (continuar.equalsIgnoreCase("s"));

        // Mostrar playlist final
        playlist.mostrarPlaylist();

        sc.close();
    }
}

import java.util.Scanner;

// Enum para tipo de postagem
enum TipoPostagem {
    TEXTO, IMAGEM, VIDEO
}

// Classe Postagem
class Postagem {
    String titulo;
    String descricao;
    TipoPostagem tipo;

    public Postagem(String titulo, String descricao, TipoPostagem tipo) {
        this.titulo = titulo;
        this.descricao = descricao;
        this.tipo = tipo;
    }
}

// Interface PlataformaSocial
interface PlataformaSocial {
    void compartilharPostagem(Postagem postagem) throws Exception;
    void compartilharImagem(Postagem postagem) throws Exception;
    void compartilharVideo(Postagem postagem) throws Exception;
}

// MyBook (aceita tudo)
class MyBook implements PlataformaSocial {

    public void compartilharPostagem(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no MyBook.");
    }

    public void compartilharImagem(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no MyBook.");
    }

    public void compartilharVideo(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no MyBook.");
    }
}

// Fotogram (não aceita texto)
class Fotogram implements PlataformaSocial {

    public void compartilharPostagem(Postagem postagem) throws Exception {
        throw new Exception("Fotogram não suporta postagens de texto.");
    }

    public void compartilharImagem(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no Fotogram.");
    }

    public void compartilharVideo(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no Fotogram.");
    }
}

// AnyTube (não aceita imagem)
class AnyTube implements PlataformaSocial {

    public void compartilharPostagem(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no AnyTube.");
    }

    public void compartilharImagem(Postagem postagem) throws Exception {
        throw new Exception("AnyTube não suporta imagens.");
    }

    public void compartilharVideo(Postagem postagem) {
        System.out.println("Você compartilhou essa postagem no AnyTube.");
    }
}

// Classe principal
public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        // Entrada do usuário
        System.out.print("Titulo: ");
        String titulo = sc.nextLine();

        System.out.print("Descrição: ");
        String descricao = sc.nextLine();

        System.out.print("Tipo (TEXTO, IMAGEM, VIDEO): ");
        TipoPostagem tipo = TipoPostagem.valueOf(sc.nextLine().toUpperCase());

        Postagem postagem = new Postagem(titulo, descricao, tipo);

        PlataformaSocial myBook = new MyBook();
        PlataformaSocial fotogram = new Fotogram();
        PlataformaSocial anyTube = new AnyTube();

            myBook.compartilharPostagem(postagem);

            System.out.println("Erro no MyBook: " + e.getMessage());
        }

        // Fotogram
        try {
            fotogram.compartilharImagem(postagem);
        } catch (Exception e) {
            System.out.println("Ocorreu um erro ao compartilhar essa postagem no Fotogram.");
        }

        // AnyTube
        try {
            anyTube.compartilharVideo(postagem);
        } catch (Exception e) {
            System.out.println("Ocorreu um erro ao compartilhar essa postagem no AnyTube.");
        }

        sc.close();
    }
}

import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

// Classe Cliente (POJO)
class Cliente {
    String nome;
    int idade;

    public Cliente(String nome, int idade) {
        this.nome = nome;
        this.idade = idade;
    }
}

// Classe Filme
class Filme {
    String nome;
    double preco;
    int idadeMinima;

    public Filme(String nome, double preco, int idadeMinima) {
        this.nome = nome;
        this.preco = preco;
        this.idadeMinima = idadeMinima;
    }
}

// Classe Ingresso
class Ingresso {
    Cliente cliente;
    Filme filme;
    String assento;

    public Ingresso(Cliente cliente, Filme filme, String assento) {
        this.cliente = cliente;
        this.filme = filme;
        this.assento = assento;
    }
}

// Classe Cinema
class Cinema {
    List<Filme> filmesDisponiveis = new ArrayList<>();
    List<Ingresso> ingressosVendidos = new ArrayList<>();

    // Buscar filme
    public Filme buscarFilme(String nome) throws Exception {
        for (Filme f : filmesDisponiveis) {
            if (f.nome.equalsIgnoreCase(nome)) {
                return f;
            }
        }
        throw new Exception("Filme não encontrado!");
    }

    // Verificar se assento está disponível
    public boolean assentoDisponivel(String assento, Filme filme) {
        for (Ingresso i : ingressosVendidos) {
            if (i.assento.equalsIgnoreCase(assento) && i.filme.nome.equalsIgnoreCase(filme.nome)) {
                return false;
            }
        }
        return true;
    }

    // Vender ingresso
    public void venderIngresso(Cliente cliente, Filme filme, String assento) throws Exception {

        if (cliente.idade < filme.idadeMinima) {
            throw new Exception("O ingresso não pode ser vendido pois sua idade não permite!");
        }

        if (!assentoDisponivel(assento, filme)) {
            throw new Exception("O ingresso não pode ser vendido pois seu assento não está mais disponível!");
        }

        Ingresso ingresso = new Ingresso(cliente, filme, assento);
        ingressosVendidos.add(ingresso);

        System.out.println("Ingresso vendido com sucesso! " +
                filme.nome + " - " + assento + " - " + cliente.nome);
    }
}

// Classe principal
public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        Cinema cinema = new Cinema();

        // Filmes disponíveis
        cinema.filmesDisponiveis.add(new Filme("Homen Aranha", 25.0, 18));
        cinema.filmesDisponiveis.add(new Filme("Vingadores", 30.0, 12));
        cinema.filmesDisponiveis.add(new Filme("Toy Story", 20.0, 0));
        cinema.filmesDisponiveis.add(new Filme("Batman", 28.0, 16));
        cinema.filmesDisponiveis.add(new Filme("Coringa", 35.0, 18));

        String continuar;

        do {
            try {
                System.out.print("\nQue filme você deseja assistir? ");
                String nomeFilme = sc.nextLine();

                System.out.print("Qual assento você deseja? ");
                String assento = sc.nextLine();

                System.out.print("Qual o seu nome? ");
                String nomeCliente = sc.nextLine();

                System.out.print("Qual a sua idade? ");
                int idade = Integer.parseInt(sc.nextLine());

                Cliente cliente = new Cliente(nomeCliente, idade);
                Filme filme = cinema.buscarFilme(nomeFilme);

                cinema.venderIngresso(cliente, filme, assento);

            } catch (Exception e) {
                System.out.println(e.getMessage());
            }

            System.out.print("Deseja comprar outro ingresso? (s/n): ");
            continuar = sc.nextLine();

        } while (continuar.equalsIgnoreCase("s"));

        sc.close();
    }
}
