import org.springframework.stereotype.Service;
import java.util.*;

@Service
public class TimeService {

    private List<Jogador> jogadores = new ArrayList<>();

    public TimeService() {
        inicializar();
    }

    private void inicializar() {
        // Titulares (0-10)
        jogadores.add(new Jogador("Goleiro", "GOL", 30));
        jogadores.add(new Jogador("Lateral Direito", "LD", 25));
        jogadores.add(new Jogador("Zagueiro 1", "ZAG", 28));
        jogadores.add(new Jogador("Zagueiro 2", "ZAG", 27));
        jogadores.add(new Jogador("Lateral Esquerdo", "LE", 26));
        jogadores.add(new Jogador("Volante", "VOL", 29));
        jogadores.add(new Jogador("Meia 1", "MEI", 24));
        jogadores.add(new Jogador("Meia 2", "MEI", 23));
        jogadores.add(new Jogador("Ponta Direita", "PD", 22));
        jogadores.add(new Jogador("Centroavante", "ATA", 31));
        jogadores.add(new Jogador("Ponta Esquerda", "PE", 21));

        // Reservas (11-15)
        jogadores.add(new Jogador("Reserva 1", "GOL", 20));
        jogadores.add(new Jogador("Reserva 2", "DEF", 22));
        jogadores.add(new Jogador("Reserva 3", "MEI", 23));
        jogadores.add(new Jogador("Reserva 4", "ATA", 24));
        jogadores.add(new Jogador("Reserva 5", "ATA", 25));
    }

    public List<Jogador> getTitulares() {
        return jogadores.subList(0, 11);
    }

    public List<Jogador> getReservas() {
        return jogadores.subList(11, jogadores.size());
    }

    public String substituirJogador(int posicaoTitular) {
        if (posicaoTitular < 0 || posicaoTitular > 10) {
            return "Posição inválida!";
        }

        if (jogadores.size() <= 11) {
            return "Sem reservas disponíveis!";
        }

        // pega o primeiro reserva
        Jogador reserva = jogadores.get(11);

        // troca
        Jogador titular = jogadores.get(posicaoTitular);
        jogadores.set(posicaoTitular, reserva);

        // remove reserva e adiciona o titular ao final da lista
        jogadores.remove(11);
        jogadores.add(titular);

        return "Substituição realizada!";
    }
}

import org.springframework.stereotype.Service;

import java.util.*;

@Service
public class BibliotecaService {

    private List<Livro> livrosDisponiveis = new ArrayList<>();
    private List<Emprestimo> emprestimos = new ArrayList<>();

    public BibliotecaService() {
        inicializar();
    }

    private void inicializar() {
        livrosDisponiveis.add(new Livro("Dom Casmurro", "Machado de Assis"));
        livrosDisponiveis.add(new Livro("1984", "George Orwell"));
        livrosDisponiveis.add(new Livro("O Hobbit", "J.R.R. Tolkien"));
        livrosDisponiveis.add(new Livro("Clean Code", "Robert C. Martin"));
    }

    public List<Livro> getLivrosDisponiveis() {
        return livrosDisponiveis;
    }

    public List<Emprestimo> getEmprestimos() {
        return emprestimos;
    }

    public String emprestarLivro(UUID livroId, String usuarioId) {
        Optional<Livro> livroOpt = livrosDisponiveis.stream()
                .filter(l -> l.getId().equals(livroId))
                .findFirst();

        if (livroOpt.isEmpty()) {
            return "Livro não disponível!";
        }

        Livro livro = livroOpt.get();

        Emprestimo emprestimo = new Emprestimo(livroId, usuarioId);

        emprestimos.add(emprestimo);
        livrosDisponiveis.remove(livro);

        return "Empréstimo realizado! ID: " + emprestimo.getEmprestimoId();
    }

    public String devolverLivro(UUID emprestimoId) {
        Optional<Emprestimo> empOpt = emprestimos.stream()
                .filter(e -> e.getEmprestimoId().equals(emprestimoId))
                .findFirst();

        if (empOpt.isEmpty()) {
            return "Empréstimo não encontrado!";
        }

        Emprestimo emprestimo = empOpt.get();

        // recria o livro (simples, pois não temos banco)
        Livro livro = new Livro("Livro devolvido", "Autor desconhecido");
        
        // mantém o mesmo ID
        try {
            java.lang.reflect.Field field = Livro.class.getDeclaredField("id");
            field.setAccessible(true);
            field.set(livro, emprestimo.getLivroId());
        } catch (Exception e) {
            e.printStackTrace();
        }

        livrosDisponiveis.add(livro);
        emprestimos.remove(emprestimo);

        return "Livro devolvido com sucesso!";
    }
}

import org.springframework.stereotype.Service;

import java.util.*;

@Service
public class RedeSocialService {

    private List<Usuario> usuarios = new ArrayList<>();

    public RedeSocialService() {
        inicializar();
    }

    private void inicializar() {
        usuarios.add(new Usuario("João", "joao@email.com"));
        usuarios.add(new Usuario("Maria", "maria@email.com"));
        usuarios.add(new Usuario("Pedro", "pedro@email.com"));
    }

    public List<Usuario> getUsuarios() {
        return usuarios;
    }

    private Optional<Usuario> buscarUsuario(UUID id) {
        return usuarios.stream()
                .filter(u -> u.getId().equals(id))
                .findFirst();
    }

    public List<Tweet> getTweets(UUID userId) {
        return buscarUsuario(userId)
                .map(Usuario::getTweets)
                .orElse(Collections.emptyList());
    }

    public String criarTweet(UUID userId, String mensagem) {
        Optional<Usuario> userOpt = buscarUsuario(userId);

        if (userOpt.isEmpty()) {
            return "Usuário não encontrado!";
        }

        Tweet tweet = new Tweet(mensagem);
        userOpt.get().getTweets().add(tweet);

        return "Tweet criado! ID: " + tweet.getTweetId();
    }

    public String atualizarTweet(UUID userId, UUID tweetId, String mensagem) {
        Optional<Usuario> userOpt = buscarUsuario(userId);

        if (userOpt.isEmpty()) {
            return "Usuário não encontrado!";
        }

        Optional<Tweet> tweetOpt = userOpt.get().getTweets().stream()
                .filter(t -> t.getTweetId().equals(tweetId))
                .findFirst();

        if (tweetOpt.isEmpty()) {
            return "Tweet não encontrado!";
        }

        tweetOpt.get().setMensagem(mensagem);
        return "Tweet atualizado!";
    }

    public String deletarTweet(UUID userId, UUID tweetId) {
        Optional<Usuario> userOpt = buscarUsuario(userId);

        if (userOpt.isEmpty()) {
            return "Usuário não encontrado!";
        }

        boolean removido = userOpt.get().getTweets()
                .removeIf(t -> t.getTweetId().equals(tweetId));

        return removido ? "Tweet removido!" : "Tweet não encontrado!";
    }
}

import org.springframework.stereotype.Service;

import java.util.*;

@Service
public class CinemaService {

    private List<Filme> filmes = new ArrayList<>();

    public List<Filme> getFilmes() {
        return filmes;
    }

    public String criarFilme(FilmeRequest request) {
        Filme filme = new Filme(
                request.getTitulo(),
                request.getGenero(),
                request.getCapacidade()
        );
        filmes.add(filme);
        return "Filme criado! ID: " + filme.getId();
    }

    public String atualizarFilme(UUID id, AtualizarFilmeRequest request) {
        Optional<Filme> filmeOpt = buscarFilme(id);

        if (filmeOpt.isEmpty()) return "Filme não encontrado!";

        Filme filme = filmeOpt.get();

        if (request.getTitulo() != null) filme.setTitulo(request.getTitulo());
        if (request.getGenero() != null) filme.setGenero(request.getGenero());
        if (request.getCapacidade() != null) filme.setCapacidade(request.getCapacidade());

        return "Filme atualizado!";
    }

    public String deletarFilme(UUID id) {
        boolean removido = filmes.removeIf(f -> f.getId().equals(id));
        return removido ? "Filme removido!" : "Filme não encontrado!";
    }

    public String comprarIngresso(UUID filmeId) {
        Optional<Filme> filmeOpt = buscarFilme(filmeId);

        if (filmeOpt.isEmpty()) return "Filme não encontrado!";

        Filme filme = filmeOpt.get();

        if (filme.getAssentosOcupados() >= filme.getCapacidade()) {
            return "Sessão lotada!";
        }

        int numeroAssento = filme.getAssentosOcupados() + 1;

        Ingresso ingresso = new Ingresso(filmeId, numeroAssento);

        filme.getIngressos().add(ingresso);
        filme.incrementarAssento();

        return "Ingresso comprado! ID: " + ingresso.getIngressoId() +
                " | Assento: " + numeroAssento;
    }

    public String devolverIngresso(UUID filmeId, UUID ingressoId) {
        Optional<Filme> filmeOpt = buscarFilme(filmeId);

        if (filmeOpt.isEmpty()) return "Filme não encontrado!";

        Filme filme = filmeOpt.get();

        Optional<Ingresso> ingressoOpt = filme.getIngressos().stream()
                .filter(i -> i.getIngressoId().equals(ingressoId))
                .findFirst();

        if (ingressoOpt.isEmpty()) return "Ingresso não encontrado!";

        filme.getIngressos().remove(ingressoOpt.get());
        filme.decrementarAssento();

        return "Ingresso devolvido!";
    }

    private Optional<Filme> buscarFilme(UUID id) {
        return filmes.stream()
                .filter(f -> f.getId().equals(id))
                .findFirst();
    }
}
